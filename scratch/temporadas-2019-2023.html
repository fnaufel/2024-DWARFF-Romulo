1  Temporadas 2019 a 2023

1.1 Ler e limpar (todos os jogadores)

Uma única tibble com todos os anos:

anos <- 2019:2023

ler_e_limpar <- function(ano) {
  
  nome <- paste0('planilhas/', ano, '_BB_DWARFF.xlsx')
  df <- readxl::read_xlsx(nome) %>% 
    rename(player = 'Player\n▲') %>% 
    select(
      -'0.0',-(média:`desvios mvp`)
    ) %>% 
    clean_names() %>% 
    mutate(
      across(
        vorp_g:dwarff_bb,
        ~ parse_number(
          .x, 
          locale = locale(decimal_mark = ",", grouping_mark = "."))
      )
    ) %>% 
    mutate(year = ano)
  
  df
  
}
df <- bind_rows(
  anos %>% 
    map(ler_e_limpar)
) %>% 
  select(year, everything()) %>% 
  arrange(year, player)
